# 08 实战示例：注意力机制

本章介绍如何使用 TileLang-Ascend 实现 Flash Attention 风格的注意力机制内核，展示了 Cube Core 和 Vector Core 协作的高级应用场景。

---

## 一、注意力机制原理

标准注意力计算公式：

```
Attention(Q, K, V) = softmax(Q @ K^T / sqrt(d)) @ V
```

其中：
- `Q`：查询矩阵，形状 `(B, H, S, D)`
- `K`：键矩阵，形状 `(B, H, S, D)`
- `V`：值矩阵，形状 `(B, H, S, D)`
- `D`：注意力头维度
- `S`：序列长度

### Flash Attention 的核心思想

标准注意力需要将完整的 `S x S` 注意力矩阵存入全局内存（O(S^2) 显存），而 Flash Attention 通过 **在线 Softmax（Online Softmax）** 技巧，将计算分块进行，只需 O(S) 的额外显存。

---

## 二、TileLang-Ascend 中的 Flash Attention

Flash Attention 是典型的 **Cube + Vector 融合算子**：
- **Cube Core**：执行 `Q @ K^T` 和 `P @ V` 矩阵乘法
- **Vector Core**：执行 softmax 的 exp、reduce_max、reduce_sum 等向量操作
- **Cube-Vector同步**：通过 `T.set_cross_flag` / `T.wait_cross_flag` 进行核间同步

### 数据流

```
Q[K_l1] ──┐                    ┌── acc_o_l0c ──> workspace_3 ─┐
           ├──> gemm_v0 ──> acc_s_l0c ──> workspace_1        │
K[K_l1] ──┘                    (Cube Core)                    │
                                                              ├──> acc_o ──> Output
acc_s_ub <── workspace_2 <── workspace_1 (after softmax)      │    (Vector Core)
     │                                                        │
     ├──> softmax (exp, reduce_max, reduce_sum, ...)         │
     │                                                        │
     └──> workspace_2 ──> gemm_v0 ──> acc_o_l0c              │
            acc_s_l1 ──┘                                      │
         V[v_l1] ──┘                                          │
```

## 三、完整实现

以下是基于 `tilelang-ascend/examples/flash_attention/flash_attn_bhsd.py` 的实现：

```python
import tilelang
from tilelang import DataType, language as T
import torch

B, S, H, D = 1, 128, 1, 512

@tilelang.jit(out_idx=[3], workspace_idx=[4, 5, 6])
def flash_attention_fwd(batch, seq_len, heads, dim):
    block_M, block_N = 64, 64
    dtype = "float16"
    accum_dtype = "float"
    sm_scale = (1.0 / dim) ** 0.5

    shape = [batch, heads, seq_len, dim]
    block_num = seq_len // block_M * heads * batch

    @T.prim_func
    def main(
        Q: T.Tensor(shape, dtype),
        K: T.Tensor(shape, dtype),
        V: T.Tensor(shape, dtype),
        Output: T.Tensor(shape, dtype),
        workspace_1: T.Tensor([block_num, block_M, block_N], accum_dtype),
        workspace_2: T.Tensor([block_num, block_M, block_N], dtype),
        workspace_3: T.Tensor([block_num, block_M, dim], accum_dtype),
    ):
        with T.Kernel(block_num, is_npu=True) as (cid, vid):
            bx = cid % (seq_len // block_M)
            by = cid // (seq_len // block_M) % heads
            bz = cid // (seq_len // block_M) // heads % batch

            # === L1 Buffer（Cube Core缓存）===
            q_l1 = T.alloc_L1([block_M, dim], dtype)
            k_l1 = T.alloc_L1([block_N, dim], dtype)
            v_l1 = T.alloc_L1([block_N, dim], dtype)
            acc_s_l1 = T.alloc_L1([block_M, block_N], dtype)

            # === L0C Buffer（矩阵乘输出寄存器）===
            acc_s_l0c = T.alloc_L0C([block_M, block_N], accum_dtype)
            acc_o_l0c = T.alloc_L0C([block_M, dim], accum_dtype)

            # === Unified Buffer（Vector Core计算缓冲）===
            VEC_NUM = 2
            acc_o = T.alloc_ub([block_M // VEC_NUM, dim], accum_dtype)
            sumexp = T.alloc_ub([block_M // VEC_NUM], accum_dtype)
            m_i = T.alloc_ub([block_M // VEC_NUM], accum_dtype)

            acc_s_ub = T.alloc_ub([block_M // VEC_NUM, block_N], accum_dtype)
            m_i_prev = T.alloc_ub([block_M // VEC_NUM], accum_dtype)
            acc_s_ub_ = T.alloc_ub([block_M // VEC_NUM, block_N], accum_dtype)
            sumexp_i_ub = T.alloc_ub([block_M // VEC_NUM], accum_dtype)
            acc_s_half = T.alloc_ub([block_M // VEC_NUM, block_N], dtype)
            acc_o_ub = T.alloc_ub([block_M // VEC_NUM, dim], accum_dtype)
            acc_o_half = T.alloc_ub([block_M // VEC_NUM, dim], dtype)

            # ==========================================
            # Cube Core 部分：矩阵乘法
            # ==========================================
            with T.Scope("C"):
                # 加载Q到L1
                T.copy(Q[bz, by, bx * block_M:(bx + 1) * block_M, :], q_l1)
                T.barrier_all()

                for k in T.serial(T.ceildiv(seq_len, block_N)):
                    # 加载K到L1
                    T.copy(K[bz, by, k * block_N:(k + 1) * block_N, :], k_l1)
                    T.barrier_all()

                    # Q @ K^T -> acc_s_l0c
                    T.gemm_v0(q_l1, k_l1, acc_s_l0c, transpose_B=True, init=True)
                    T.barrier_all()

                    # L0C -> workspace_1 (GM)
                    T.copy(acc_s_l0c, workspace_1[cid, :, :])
                    T.barrier_all()

                    # 通知Vector：attention score已写入workspace
                    T.set_cross_flag("FIX", 0)
                    T.wait_cross_flag(1)
                    T.barrier_all()

                    # 加载softmax后的score和V
                    T.copy(workspace_2[cid, :, :], acc_s_l1)
                    T.copy(V[bz, by, k * block_N:(k + 1) * block_N, :], v_l1)
                    T.barrier_all()

                    # P @ V -> acc_o_l0c
                    T.gemm_v0(acc_s_l1, v_l1, acc_o_l0c, init=True)
                    T.barrier_all()

                    # L0C -> workspace_3 (GM)
                    T.copy(acc_o_l0c, workspace_3[cid, :, :])
                    T.barrier_all()

                    T.set_cross_flag("FIX", 2)
                    T.wait_cross_flag(3)

            # ==========================================
            # Vector Core 部分：在线Softmax
            # ==========================================
            with T.Scope("V"):
                # 初始化累加器
                T.tile.fill(acc_o, 0.0)
                T.tile.fill(sumexp, 0.0)
                T.tile.fill(m_i, -2 ** 30)
                T.barrier_all()

                for _k in T.serial(T.ceildiv(seq_len, block_N)):
                    T.tile.fill(acc_s_ub, 0.0)
                    T.barrier_all()

                    # 保存上一轮的m_i
                    T.copy(m_i, m_i_prev)
                    T.barrier_all()

                    # 等待Cube写入attention score
                    T.wait_cross_flag(0)
                    T.copy(
                        workspace_1[cid, vid * block_M // VEC_NUM:vid * block_M // VEC_NUM + block_M // VEC_NUM, :],
                        acc_s_ub_
                    )
                    T.barrier_all()

                    # 累加score
                    T.tile.add(acc_s_ub, acc_s_ub, acc_s_ub_)
                    T.barrier_all()

                    # 乘以scale
                    T.tile.mul(acc_s_ub, acc_s_ub, sm_scale)
                    T.barrier_all()

                    # 计算新的行最大值
                    T.reduce_max(acc_s_ub, m_i, dim=-1)
                    T.barrier_all()

                    # m_i = max(m_i_prev, m_i)
                    T.tile.max(m_i, m_i, m_i_prev)
                    T.barrier_all()

                    # 修正因子：exp(m_i_prev - m_i)
                    T.tile.sub(m_i_prev, m_i_prev, m_i)
                    T.barrier_all()
                    T.tile.exp(m_i_prev, m_i_prev)
                    T.barrier_all()

                    # score减去行最大值
                    for h_i in range(block_M // VEC_NUM):
                        T.barrier_all()
                        T.tile.sub(acc_s_ub[h_i, :], acc_s_ub[h_i, :], m_i[h_i])
                        T.barrier_all()

                    # exp(score - max)
                    T.tile.exp(acc_s_ub, acc_s_ub)
                    T.barrier_all()

                    # sum(exp)
                    T.reduce_sum(acc_s_ub, sumexp_i_ub, dim=-1)
                    T.barrier_all()

                    # 更新sumexp: sumexp = sumexp * correction + sumexp_i
                    T.tile.mul(sumexp, sumexp, m_i_prev)
                    T.barrier_all()
                    T.tile.add(sumexp, sumexp, sumexp_i_ub)
                    T.barrier_all()

                    # 更新acc_o: acc_o = acc_o * correction
                    for h_i in range(block_M // VEC_NUM):
                        T.barrier_all()
                        T.tile.mul(acc_o[h_i, :], acc_o[h_i, :], m_i_prev[h_i])
                        T.barrier_all()

                    # 将softmax后的score转为float16写入workspace
                    T.copy(acc_s_ub, acc_s_half)
                    T.barrier_all()
                    T.copy(acc_s_half, workspace_2[cid, vid * block_M // VEC_NUM:vid * block_M // VEC_NUM + block_M // VEC_NUM, :])
                    T.barrier_all()

                    # 通知Cube：score已准备好
                    T.set_cross_flag("MTE3", 1)

                    # 等待Cube完成PV矩阵乘
                    T.wait_cross_flag(2)
                    T.barrier_all()

                    # 累加到输出: acc_o += workspace_3
                    T.copy(workspace_3[cid, vid * block_M // VEC_NUM:vid * block_M // VEC_NUM + block_M // VEC_NUM, :], acc_o_ub)
                    T.barrier_all()
                    T.tile.add(acc_o, acc_o, acc_o_ub)
                    T.barrier_all()

                    T.set_cross_flag("V", 3)
                    T.barrier_all()

                # 最终归一化: acc_o /= sumexp
                for h_i in range(block_M // VEC_NUM):
                    T.barrier_all()
                    T.tile.div(acc_o[h_i, :], acc_o[h_i, :], sumexp[h_i])
                    T.barrier_all()

                # 类型转换和写回
                T.copy(acc_o, acc_o_half)
                T.barrier_all()
                T.copy(
                    acc_o_half,
                    Output[bz, by, bx * block_M + vid * block_M // VEC_NUM:bx * block_M +
                           vid * block_M // VEC_NUM + block_M // VEC_NUM, :]
                )

    return main
```

## 四、运行与验证

```python
if __name__ == "__main__":
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument("--B", type=int, default=1)
    parser.add_argument("--S", type=int, default=128)
    parser.add_argument("--H", type=int, default=1)
    parser.add_argument("--D", type=int, default=512)
    args = parser.parse_args()
    B, S, H, D = args.B, args.S, args.H, args.D

    torch.set_default_device('npu')
    torch.manual_seed(0)
    tilelang.disable_cache()

    func = flash_attention_fwd(batch=B, seq_len=S, heads=H, dim=D)

    # 准备数据
    q = torch.randn((B, H, S, D), dtype=torch.float16)
    k = torch.randn((B, H, S, D), dtype=torch.float16)
    v = torch.randn((B, H, S, D), dtype=torch.float16)

    torch.npu.synchronize()
    print("init successful!")

    # TileLang执行
    output = func(q, k, v)

    # 参考实现
    def ref_flash_attn(q, k, v):
        q = q.float()
        k = k.float()
        v = v.float()
        acc = torch.einsum("bhsd,bhkd->bhsk", q, k) * (1.0 / q.shape[-1]) ** 0.5
        acc = acc.softmax(dim=-1)
        o = torch.einsum("bhsk,bhkd->bhsd", acc, v)
        return o.to(torch.float16)

    ref_output = ref_flash_attn(q, k, v)
    torch.npu.synchronize()

    torch.testing.assert_close(ref_output, output, rtol=1e-2, atol=1e-2)
    print("Test Passed!")
```

## 五、关键技术点

### 5.1 Cube-Vector核间同步

Flash Attention的核心难点是Cube Core和Vector Core之间的数据传递和同步：

```python
# Cube端：计算完成后通知Vector
T.set_cross_flag("FIX", 0)  # 设置标志0

# Vector端：等待Cube完成
T.wait_cross_flag(0)         # 等待标志0
# ... 处理数据 ...
T.set_cross_flag("MTE3", 1)  # 处理完成后通知Cube

# Cube端：等待Vector完成
T.wait_cross_flag(1)          # 等待标志1
```

### 5.2 Workspace机制

Cube和Vector之间通过全局内存中的workspace传递数据：

```python
# Cube -> workspace_1 -> Vector
T.copy(acc_s_l0c, workspace_1[cid, :, :])    # Cube写
T.copy(workspace_1[cid, ...], acc_s_ub_)      # Vector读

# Vector -> workspace_2 -> Cube
T.copy(acc_s_half, workspace_2[cid, ...])     # Vector写
T.copy(workspace_2[cid, :, :], acc_s_l1)      # Cube读
```

workspace通过 `workspace_idx` 参数声明：

```python
@tilelang.jit(out_idx=[3], workspace_idx=[4, 5, 6])
```

### 5.3 在线Softmax算法

```python
# 初始化
T.tile.fill(acc_o, 0.0)       # O = 0
T.tile.fill(sumexp, 0.0)      # l = 0
T.tile.fill(m_i, -2**30)      # m = -inf

# 每轮迭代
# 1. 计算新的最大值
T.reduce_max(acc_s_ub, m_i, dim=-1)
T.tile.max(m_i, m_i, m_i_prev)    # m_new = max(m_old, rowmax(S))

# 2. 计算修正因子
T.tile.sub(m_i_prev, m_i_prev, m_i)
T.tile.exp(m_i_prev, m_i_prev)    # correction = exp(m_old - m_new)

# 3. 修正之前的累加
T.tile.mul(sumexp, sumexp, m_i_prev)  # l = l * correction
T.tile.mul(acc_o, acc_o, m_i_prev)    # O = O * correction

# 4. 更新
T.tile.exp(acc_s_ub, acc_s_ub)        # P = exp(S - m_new)
T.reduce_sum(acc_s_ub, sumexp_i_ub, dim=-1)
T.tile.add(sumexp, sumexp, sumexp_i_ub)  # l = l + sum(P)

# 最终归一化
T.tile.div(acc_o[h_i, :], acc_o[h_i, :], sumexp[h_i])  # O = O / l
```

## 六、性能优化要点

| 优化项 | 说明 |
|--------|------|
| 分块大小 | block_M=64, block_N=64 适合大多数场景 |
| Cube-Vector流水线 | 使用 `T.Pipelined` + 核间流水线隐藏延迟 |
| VEC_NUM=2 | 充分利用每个AI Core的2个Vector单元 |
| workspace复用 | 多个workspace可通过地址规划复用空间 |
| 自动CV分离 | 开启 `TL_ASCEND_AUTO_CV_COMBINE` 简化开发 |

## 总结

本章展示了如何使用 TileLang-Ascend 实现高性能的 Flash Attention 内核。关键要点：

1. **Cube+Vector协作** 是昇腾NPU上融合算子的核心模式
2. **Workspace机制** 用于Cube和Vector之间的数据传递
3. **核间同步** 通过 `T.set_cross_flag` / `T.wait_cross_flag` 实现
4. **在线Softmax** 是 Flash Attention 的核心算法，在Vector Core上执行
5. **VEC_NUM=2** 需要将Tile均分给两个Vector单元
